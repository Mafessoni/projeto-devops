CREATE TABLE IF NOT EXISTS `ENTIDADES` (
	ID_ENTIDADES INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
	NOME VARCHAR(150),
	RAZAOSOCIAL VARCHAR(200),
	CNPJCPF VARCHAR(20),
	ENDERECO VARCHAR(150),
	BAIRRO VARCHAR(150),
	NUMERO INT,
	CEP VARCHAR(20),
	CIDADE VARCHAR(150),
	ESTADO VARCHAR(2)
	
);

CREATE TABLE IF NOT EXISTS `OPERADORES` (
	ID_OPERADORES INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	NOME VARCHAR(100),
	SENHA VARCHAR(100),
	ATIVO TINYINT
);
